% nadpis bez čísel
\newcommand{\nadpisBezCisla}[1]{
	\newpage
    \setcounter{chapter}{0}
    \setcounter{section}{0}
    \chapter*{#1}
    \addcontentsline{toc}{chapter}{#1}
}

\newcommand{\podnadpisBezCisla}[1]{
    \setcounter{chapter}{0}
    \setcounter{section}{0}
    \section*{#1}
    \addcontentsline{toc}{section}{#1}
}
